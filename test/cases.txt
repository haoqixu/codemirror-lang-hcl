# Empty file

==>

ConfigFile()

# Single attribute

one = two

==>

ConfigFile(Body(Attribute(Identifier, Expression(VariableExpr(Identifier)))))

# Attributes

one = two
type = string
key = val

==>

ConfigFile(Body(
        Attribute(Identifier, Expression(VariableExpr(Identifier))),
        Attribute(Identifier, Expression(VariableExpr(Identifier))),
        Attribute(Identifier, Expression(VariableExpr(Identifier)))))

# Literal null

a = null
null1 = null
null2 = null
null = null

==>

ConfigFile(Body(
        Attribute(Identifier, Expression(LiteralValue(NullLit(null)))),
        Attribute(Identifier, Expression(LiteralValue(NullLit(null)))),
        Attribute(Identifier, Expression(LiteralValue(NullLit(null)))),
        Attribute(Identifier, Expression(LiteralValue(NullLit(null))))))
