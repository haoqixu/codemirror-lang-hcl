# Template interpolation

a = "${hello}abc"

==>

ConfigFile(
        Body(
            Attribute(
                Identifier
                Expression(
                    TemplateExpr(
                        QuotedTemplate(
                            QuotedTemplateStart
                            TemplateInterpolation(
                                TemplateInterpolationStart
                                Expression(VariableExpr(Identifier))
                                TemplateInterpolationEnd)
                            TemplateLiteral,
                            QuotedTemplateEnd))))))

# Simple Heredoc

f = <<-EOF
  hello, EO
  F
  world, EOF
  EOF xxx
 EOF
a = b

==>

ConfigFile(
        Body(
            Attribute(
                Identifier
                Expression(
                    TemplateExpr(
                        HeredocTemplate(
                            HeredocStart
                            HeredocIdentifier
                            TemplateLiteral
                            HeredocIdentifier))))
            Attribute(
                Identifier
                Expression(
                    VariableExpr(Identifier)
                )
            )))
